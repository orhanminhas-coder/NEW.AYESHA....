<script>
// Countdown Timer (3 sec)
let countdown = 3;
let countdownEl = document.getElementById("countdown");
let interval = setInterval(()=>{
countdown--;
countdownEl.innerHTML = "Surprise in " + countdown + " seconds...";
if(countdown <= 0){
clearInterval(interval);
countdownEl.innerHTML = "🎂 It's time! 🎂";
createHearts();
}
},1000);

// Photo Cube rotation handled by CSS animation

// Stars Background
let canvas = document.getElementById("stars");
let ctx = canvas.getContext("2d");
canvas.width = window.innerWidth;
canvas.height = window.innerHeight;
let stars = [];
for(let i=0;i<300;i++){
stars.push({x:Math.random()*canvas.width, y:Math.random()*canvas.height, r:Math.random()*2});
}
function drawStars(){
ctx.clearRect(0,0,canvas.width,canvas.height);
ctx.fillStyle="white";
stars.forEach(s=>{
ctx.beginPath();
ctx.arc(s.x,s.y,s.r,0,Math.PI*2);
ctx.fill();
});
requestAnimationFrame(drawStars);
}
drawStars();

// Heart rain
function createHearts(){
setInterval(()=>{
let heart = document.createElement("div");
heart.classList.add("heart");
heart.innerHTML="❤️";
heart.style.left = Math.random()*100+"vw";
heart.style.animationDuration = (Math.random()*3+2)+"s";
document.body.appendChild(heart);
setTimeout(()=>heart.remove(),5000);
},200);
}

// Fireworks
function createFireworks(){
for(let f=0; f<60; f++){
let fire = document.createElement("div");
fire.innerHTML="✨";
fire.style.position="absolute";
fire.style.color=["yellow","pink","red"][Math.floor(Math.random()*3)];
fire.style.left=(window.innerWidth/2 + (Math.random()*200-100))+"px";
fire.style.top=(window.innerHeight/2 + (Math.random()*200-100))+"px";
fire.style.fontSize=(Math.random()*20+10)+"px";
document.body.appendChild(fire);
setTimeout(()=>fire.remove(),1500);
}
}

// Gift Box Click
function openGift(){
createFireworks();
document.getElementById("card").style.display="block";
typingEffect("Happy Birthday Ayesha ❤️ You are the most special person in my life. May your day shine with love, happiness, and dreams coming true. 🌹");
}

// Typing Effect
function typingEffect(text){
let i=0;
let box = document.getElementById("typedMessage");
box.innerHTML="";
let interval = setInterval(()=>{
box.innerHTML += text[i];
i++;
if(i>=text.length) clearInterval(interval);
},50);
}
